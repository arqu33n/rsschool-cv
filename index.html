function parse(data) {
let result = [];
let counter = 0;

  data.split("").forEach((item) => {
    if (item == "i") {
    counter++;
    } else if (item == "d") {
    counter--;
    } else if (item == "s") {
    counter **= 2;
    } else if (item == "o") {
    result.push(counter);
    } else return;
  });
return result;
}
